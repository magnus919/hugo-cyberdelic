{{- /* pagination.html — Next/previous article navigation Shows chronologically adjacent articles with their titles. Controlled by params.showPagination. First article: no next (newer) link. Last article: no previous (older) link. Single-article sections: no links at all. rel=prev/next for SEO, accessible focus indicators via base styles. */ -}} {{- $page := . -}} {{- $showPagination := $page.Site.Params.showPagination | default false -}} {{- if $showPagination -}} {{- $prev := $page.PrevInSection -}} {{- $next := $page.NextInSection -}} {{- if or $prev $next -}}
{{- with $prev -}} {{ i18n "pagination.previous" | default "Previous" }} {{ .Title }} {{- end -}} {{- with $next -}} {{ i18n "pagination.next" | default "Next" }} {{ .Title }} {{- end -}}
{{- end -}} {{- end -}}

{{- /* alert.html — GitHub-flavored alert shortcode Parameters: type (string, optional) — Alert variant: note, tip, important, warning, caution. Defaults to "note" if missing or invalid. Body: Arbitrary markdown rendered inside the alert container. Each variant has a distinct icon, color, and ARIA role. - note / tip: role="note" - important / warning / caution: role="alert" */ -}} {{- $type := .Get "type" | default "note" | lower -}} {{- /* Validate type; fall back to "note" if unrecognized */ -}} {{- $validTypes := slice "note" "tip" "important" "warning" "caution" -}} {{- if not (in $validTypes $type) -}} {{- $type = "note" -}} {{- end -}} {{- /* Map variant to ARIA role */ -}} {{- $role := "note" -}} {{- if or (eq $type "important") (eq $type "warning") (eq $type "caution") -}} {{- $role = "alert" -}} {{- end -}} {{- /* Heading label from i18n */ -}} {{- $label := i18n (printf "alert.%s" $type) | default ($type | title) -}} {{- /* SVG icon definitions per variant */ -}} {{- $icon := "" -}} {{- if eq $type "note" -}} {{- $icon = `` -}} {{- else if eq $type "tip" -}} {{- $icon = `` -}} {{- else if eq $type "important" -}} {{- $icon = `` -}} {{- else if eq $type "warning" -}} {{- $icon = `` -}} {{- else if eq $type "caution" -}} {{- $icon = `` -}} {{- end -}}
{{ $icon | safeHTML }} {{ $label }}
